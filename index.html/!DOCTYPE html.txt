    <script>
        // Define the main celebration function
        function revealCelebration() {
            // 1. Find the hidden gallery section using its ID
            const gallery = document.getElementById('photoGallery');
            // 2. Find the button using its ID
            const button = document.getElementById('surpriseButton');

            // 3. Update the button appearance and disable it
            button.innerText = "Surprise Activated! ✨";
            button.classList.remove('bg-pink-500', 'hover:bg-pink-600');
            button.classList.add('bg-gray-400', 'cursor-not-allowed');
            button.disabled = true;

            // 4. Reveal the gallery
            gallery.classList.remove('hidden'); // Make it exist in the layout
            gallery.classList.add('gallery-reveal'); // Trigger the smooth animation

            // 5. Fire multiple confetti bursts for maximum effect
            // Left burst
            confetti({
                particleCount: 150,
                spread: 80,
                origin: { x: 0.1, y: 0.7 }
            });
            // Right burst
            confetti({
                particleCount: 150,
                spread: 80,
                origin: { x: 0.9, y: 0.7 }
            });
            // Final center burst
            setTimeout(() => {
                confetti({
                    particleCount: 100,
                    spread: 100,
                    origin: { x: 0.5, y: 0.5 }
                });
            }, 300);
        }
    </script>
